name: 自动更新 IP 列表

on:
  schedule:
    # 每天北京时间凌晨 5:00 (UTC 前一天 21:00)
    - cron: "0 21 * * *"
  workflow_dispatch:

jobs:
  update-ip-list:
    runs-on: ubuntu-latest

    steps:
      - name: 检出仓库代码
        uses: actions/checkout@v4

      - name: 设置 Python 环境
        uses: actions/setup-python@v5
        with:
          python-version: "3.11"

      - name: 安装依赖
        run: |
          pip install -r requirements.txt || true
          pip install requests beautifulsoup4 geoip2

      - name: 运行更新脚本
        run: python ./yxip/collect_ips_old.py

      - name: 提交更新结果
        run: |
          git config --global user.name "github-actions[bot]"
          git config --global user.email "github-actions[bot]@users.noreply.github.com"
          git add ./yxip/ip.txt || true
          git commit -m "自动更新 IP 列表 $(date '+%Y-%m-%d %H:%M:%S')" || echo "没有变化"
          git push
